<program> ::= <expr>*
<expr> ::= <term>((PLUS|MINUS) term)* |
                                        KEYWORD:VAR IDENTIFIER EQ expr
<term> ::= <factor> ((MUL|DIV) factor)*
<factor> ::= <number>|<identifier> |
             (PLUS|MINUS) factor |
              LPAREN expr RPAREN
<number> ::= [0-9.]* ;only one dot
<identifier> ::= [a-z]*
